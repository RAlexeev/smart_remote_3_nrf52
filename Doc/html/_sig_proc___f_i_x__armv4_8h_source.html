33 static OPUS_INLINE opus_int32 silk_MLA_armv4(opus_int32 a, opus_int32 b,
34  opus_int32 c)
35 {
36  opus_int32 res;
37 
38 #if defined( __CC_ARM )
39  __asm{ MLA res, b, c, a }
40 #elif defined( __ICCARM__ )
41  __asm(
42  "mla %0, %1, %2, %3\n\t"
43  : "=&r"(res)
44  : "r"(b), "r"(c), "r"(a)
45  );
46 #else
47  __asm__(
48  "#silk_MLA\n\t"
49  "mla %0, %1, %2, %3\n\t"
50  : "=&r"(res)
51  : "r"(b), "r"(c), "r"(a)
52  );
53 #endif
54 
55  return res;
56 }

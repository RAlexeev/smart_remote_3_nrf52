185 static OPUS_INLINE int toOpus(unsigned char c)
186 {
187  int ret=0;
188  if (c<0xA0)
189  ret = toOpusTable[c>>3];
190  if (ret == 0)
191  return -1;
192  else
193  return ret|(c&0x7);
194 }
195 
196 static OPUS_INLINE int fromOpus(unsigned char c)
197 {
198  if (c<0x80)
199  return -1;
200  else
201  return fromOpusTable[(c>>3)-16] | (c&0x7);
202 }
